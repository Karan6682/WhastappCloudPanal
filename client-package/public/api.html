                    POST
                    
Send PDF Document

                

                
            

            

                

                    
API Endpoint

                    

                        
                        
POST http://localhost:3001/webhook/send-pdf

                    

                

                
                

                    
Headers

                    

                        
                        
X-API-Key: whatsapp_api_key_2024_secure

                    

                

                
                

                    
Request Body (multipart/form-data)

                    

                        
phone: 918683916682
pdf: [PDF File]
caption: Here is your document (optional)

                    

                

                
                

                    
Parameters

                    

                        
ParameterTypeDescription

                        
phone*RequiredStringPhone number with country code

                        
pdf*RequiredFilePDF file to send (max 50MB)

                        
captionOptionalStringCaption for the PDF

                        
sessionOptionalStringSession name (default: admin)

                    

                

                
                

                    
                    
                    
                    
                

                
                

                    

                        
                        
curl -X POST http://localhost:3001/webhook/send-pdf \
  -H "X-API-Key: whatsapp_api_key_2024_secure" \
  -F "phone=918683916682" \
  -F "pdf=@document.pdf" \
  -F "caption=Here is your document"

                    

                

                
                

                    

                        
                        
import requests

url = "http://localhost:3001/webhook/send-pdf"
headers = {"X-API-Key": "whatsapp_api_key_2024_secure"}

files = {"pdf": open("document.pdf", "rb")}
data = {
    "phone": "918683916682",
    "caption": "Here is your document"
}

response = requests.post(url, headers=headers, files=files, data=data)
print(response.json())

                    

                

                
                

                    

                        
                        
using System.Net.Http;

var client = new HttpClient();
client.DefaultRequestHeaders.Add("X-API-Key", "whatsapp_api_key_2024_secure");

var form = new MultipartFormDataContent();
form.Add(new StringContent("918683916682"), "phone");
form.Add(new StringContent("Here is your document"), "caption");
form.Add(new ByteArrayContent(File.ReadAllBytes("document.pdf")), "pdf", "document.pdf");

var response = await client.PostAsync("http://localhost:3001/webhook/send-pdf", form);
var result = await response.Content.ReadAsStringAsync();
Console.WriteLine(result);
